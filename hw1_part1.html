def age_category(age):
    '''
    < 30 -> 1
    >= 30, <55 -> 2
    >= 55 -> 3
    '''
    if age < 30:
        return 1
    elif age < 55:
        return 2
    else:
        return 3

def first_name(name):
    import re
    # регулярка, ищущая первое слово, идущее сразу после 'Mr.' или 'Mrs.'
    return re.search(r'(?<=\b[Mr\.|Mrs\.]\s)(\w+)|$', name).group()
    

data['First_name'] = data['Name'].apply(first_name)
